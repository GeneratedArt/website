5. Data Model (D1)
Start with this schema. Use Drizzle for types. Generate migrations; every migration runs in CI.
sqlCREATE TABLE users (
  id              TEXT PRIMARY KEY,         -- ulid
  github_login    TEXT UNIQUE,
  wallet_address  TEXT UNIQUE,              -- lowercased, 0x...
  display_name    TEXT,
  bio             TEXT,
  avatar_r2_key   TEXT,
  role            TEXT NOT NULL DEFAULT 'collector', -- collector|artist|gallery|curator|steward
  created_at      INTEGER NOT NULL
);

CREATE TABLE artists (
  user_id         TEXT PRIMARY KEY REFERENCES users(id),
  slug            TEXT UNIQUE NOT NULL,
  website         TEXT,
  socials_json    TEXT,                     -- {twitter, instagram, ...}
  status          TEXT NOT NULL DEFAULT 'pending', -- pending|approved|suspended
  approved_at     INTEGER,
  approved_by     TEXT REFERENCES users(id)
);

CREATE TABLE projects (
  id              TEXT PRIMARY KEY,
  artist_id       TEXT NOT NULL REFERENCES users(id),
  slug            TEXT UNIQUE NOT NULL,
  title           TEXT NOT NULL,
  description     TEXT,
  github_repo     TEXT NOT NULL,            -- 'GeneratedArt/art-<slug>'
  release_tag     TEXT,                     -- 'v1.0.0', matches immutable git tag
  bundle_cid      TEXT,                     -- IPFS CID of the zipped bundle
  license         TEXT NOT NULL DEFAULT 'CC-BY-NC-4.0',
  edition_size    INTEGER NOT NULL,
  price_wei       TEXT NOT NULL,            -- stringified uint256
  royalty_bps     INTEGER NOT NULL DEFAULT 750, -- 7.5%
  contract_address TEXT,                    -- set after factory deploy
  status          TEXT NOT NULL DEFAULT 'draft', -- draft|review|approved|live|sold_out|archived
  created_at      INTEGER NOT NULL,
  minted_count    INTEGER NOT NULL DEFAULT 0
);

CREATE TABLE editions (
  id              TEXT PRIMARY KEY,         -- contract_address:token_id
  project_id      TEXT NOT NULL REFERENCES projects(id),
  token_id        INTEGER NOT NULL,
  token_hash      TEXT NOT NULL,            -- 0x... 32-byte seed
  owner_address   TEXT NOT NULL,
  minted_at       INTEGER NOT NULL,
  preview_r2_key  TEXT,
  metadata_cid    TEXT,
  UNIQUE(project_id, token_id)
);

CREATE TABLE galleries (
  id              TEXT PRIMARY KEY,
  slug            TEXT UNIQUE NOT NULL,
  name            TEXT NOT NULL,
  city            TEXT,
  country         TEXT,
  description     TEXT,
  steward_user_id TEXT REFERENCES users(id),
  physical        INTEGER NOT NULL DEFAULT 0 -- bool
);

CREATE TABLE exhibitions (
  id              TEXT PRIMARY KEY,
  gallery_id      TEXT NOT NULL REFERENCES galleries(id),
  title           TEXT NOT NULL,
  starts_at       INTEGER NOT NULL,
  ends_at         INTEGER,
  project_ids_json TEXT NOT NULL            -- JSON array
);

CREATE TABLE physical_editions (
  edition_id      TEXT PRIMARY KEY REFERENCES editions(id),
  dongle_serial   TEXT UNIQUE,
  print_spec_json TEXT,                     -- {paper, size, plotter, ink}
  shipped_to      TEXT,                     -- encrypted address
  status          TEXT NOT NULL DEFAULT 'pending'
);

CREATE TABLE sessions (
  token           TEXT PRIMARY KEY,
  user_id         TEXT NOT NULL REFERENCES users(id),
  expires_at      INTEGER NOT NULL
);

CREATE TABLE audit_log (
  id              INTEGER PRIMARY KEY AUTOINCREMENT,
  actor_user_id   TEXT,
  action          TEXT NOT NULL,
  subject         TEXT,
  details_json    TEXT,
  created_at      INTEGER NOT NULL
);
